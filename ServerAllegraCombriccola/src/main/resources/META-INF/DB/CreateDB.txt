	IDEVENTO BIGINT auto_increment NOT NULL,
	NOME varchar(200) NOT NULL,
	URL varchar(400) NULL COMMENT 'URL DEL SITO WEB DELL'' EVENTO',
	INIZIOEVENTO TIMESTAMP NOT NULL COMMENT 'DATA E ORARIO INIZIO DELL'' EVENTO',
	FINEEVENTO TIMESTAMP NOT NULL COMMENT 'DATA E ORARIO FINE EVENTO',
	VIA varchar(600) NOT NULL COMMENT 'VIA COMPRESO CIVICO',
	CITTA varchar(300) NOT NULL,
	CAP varchar(5) NOT NULL,
	PROVINCIA varchar (100) NOT NULL,
	REGIONE varchar (30) NOT NULL,

	IMMAGINEEVENTO varchar(300) NULL COMMENT 'NOME IMMAGINE DELL'' EVENTO, CHE SARA'' CARICATA AD DI UN PATH DEFINITO',
	CONSTRAINT EVENTO_pk PRIMARY KEY (IDEVENTO)
);
