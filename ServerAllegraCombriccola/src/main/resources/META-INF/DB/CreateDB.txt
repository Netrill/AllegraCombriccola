CREATE TABLE allegracombriccoladb.EVENTO (
	IDEVENTO BIGINT auto_increment NOT NULL,
	NOME varchar(200) NOT NULL,
	URL varchar(400) NULL COMMENT 'URL DEL SITO WEB DELL'' EVENTO',
	INIZIOEVENTO TIMESTAMP NOT NULL,
	FINEEVENTO TIMESTAMP NOT NULL,
	VIA varchar(600) NOT NULL COMMENT 'VIA COMPRESO CIVICO',
	CITTA varchar(300) NOT NULL,
	CAP varchar (10) NOT NULL,
	PROVINCIA varchar(100) NOT NULL,
	REGIONE varchar(100) NOT NULL,
	DESCRIZIONE varchar (5000) NOT NULL,
	IMMAGINIEVENTO varchar(3000) NULL COMMENT 'NOME IMMAGINE DELL'' EVENTO, CHE SARA'' CARICATA AD DI UN PATH DEFINITO',
	LNG varchar(30) NOT NULL DEFAULT 0,
	LAT varchar(30) NOT NULL DEFAULT 0,
	SALVATO boolean DEFAULT false,
	CONSTRAINT EVENTO_pk PRIMARY KEY (IDEVENTO)
);
